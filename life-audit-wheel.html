  <script>

    let chart;

    function generateChart() {

      const labels = [

        'Health & Wellbeing', 'Finances', 'Career & Impact', 'Relationships',

        'Personal Growth', 'Fun & Creativity', 'Environment', 'Spirituality or Purpose'

      ];

      const values = [];

      for (let i = 0; i < labels.length; i++) {

        values.push(parseInt(document.getElementById(`cat${i}`).value) || 0);

      }

      if (chart) chart.destroy();

      const ctx = document.getElementById('lifeWheel').getContext('2d');

      chart = new Chart(ctx, {

        type: 'radar',

        data: {

          labels: labels,

          datasets: [{

            label: 'Your Life Audit',

            data: values,

            backgroundColor: 'rgba(54, 162, 235, 0.2)',

            borderColor: 'rgba(54, 162, 235, 1)',

            borderWidth: 2,

            pointBackgroundColor: 'rgba(54, 162, 235, 1)',

          }]

        },

        options: {

          scales: {

            r: {

              min: 0,

              max: 10,

              ticks: { stepSize: 2 }

            }

          }

        }

      });

    }

    function downloadPDF() {

      html2canvas(document.getElementById('chart-container')).then(canvas => {

        const imgData = canvas.toDataURL('image/png');

        const { jsPDF } = window.jspdf;

        const pdf = new jsPDF();

        pdf.text("Your Life Audit Wheel", 20, 20);

        pdf.addImage(imgData, 'PNG', 15, 30, 180, 160);

        pdf.save('LifeAuditWheel.pdf');

      });

    }

  </script>
